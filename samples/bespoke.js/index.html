Let's make a progress bar...


bespoke.plugins.progress = function(deck) {

 var progress = document.querySelector('#progress');

  deck.on('activate', function(event) {
    progress.style.width = (
        (event.index * 100) /
        (deck.slides.length - 1)
      ) + '%';
  });

};
